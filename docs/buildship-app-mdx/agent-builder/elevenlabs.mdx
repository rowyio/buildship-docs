export const formatInputsToTable = (schema) => {
  const properties = schema?.properties || {};

  const sortedProps = Object.entries(properties).sort((a, b) => {
    const indexA = a[1].buildship?.index ?? Infinity;
    const indexB = b[1].buildship?.index ?? Infinity;
    return indexA - indexB;
  });

  return (
    <Box sx={{
      marginTop: "12px",
    }}>
      <Box sx={{
        display: "flex",
        flexDirection: "row",
        flexGrow: 1,
      }}>
        <Box sx={{ flex: "1 0 25%" }}>
          <Typography
            sx={(theme) => ({
              color: theme.palette.text[400],
              ...theme.typography.body,
            })}
          >
            Data type
          </Typography>
        </Box>
        <Box sx={{ flex: "1 0 75%", width: "75%", }}>
          <Typography
            sx={(theme) => ({
              color: theme.palette.text[400],
              ...theme.typography.body,
            })}
          >
            Identifier
          </Typography>
        </Box>
      </Box>

      {sortedProps.map(([key, value]) => (
        <Box key={key} sx={{
          display: "flex",
          flexDirection: "column",
          flexGrow: 1,
          gap: "2px",
          marginBottom: "8px",
        }}>
          <Box sx={{
            display: "flex",
            flexDirection: "row",
            flexGrow: 1,
            gap: "8px",
            minHeight: "28px",
          }}>
            <Box sx={{ flex: "1 0 20%" }}>
              <ClipboardField
                content={value.type.charAt(0).toUpperCase() + value.type.slice(1)}
                showCopyButton={false}
              />
            </Box>
            <Box sx={{ flex: "1 0 75%", width: "75%", }}>
              <ClipboardField content={key} />
            </Box>
          </Box>

          <Box sx={{ flex: "1", width: "100%", }}>
            <ClipboardField
              content={value.description || '[No description provided]'}
              description=""
              multiline={false}
            />
          </Box>
        </Box>
      ))}
    </Box>
  );
};
